char replace(char sentence){
	char help[1000];
	int index = 0;
    while( sentence = getchar(), sentence >= 0 ){
		if(sentence[index] == ' '){
			help[index] = '-';
		}
		else{
			help[index] = sentence[index];
		}
		index++;
	}
	return help;
}
