int number;
int step;
main(){
	scanf("%d", &number);
	printf("initial value is %d", &number);
	step = 0;
	while(number>1){
		if(number <1){
			printf("Error");
			exit(0);
		}
		else if(number%2 == 0){
			number = number/2;
			step++;
		}
		else{
			number = (number*3)+1;
			step++;
		}
		printf("next value is %d", &number);
	}
	printf("final value %d , number of steps %d", number, step);
}
